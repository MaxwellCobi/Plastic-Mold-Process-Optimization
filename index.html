Define Objective Function and Constraints

In[11]:
# Objective function: Minimize Cycle time
def objective(x):
    return predict_cycle_time(x)

# Constraints
constraints = [
    {'type': 'ineq', 'fun': lambda x: predict_shot_volume(x) - 18.7},  # Shot volume >= 18.7
    {'type': 'ineq', 'fun': lambda x: 950 - predict_clamp_force(x)},  # Clamp force <= 950
    {'type': 'ineq', 'fun': lambda x: 130 - predict_torque_peak(x)}   # Torque peak <= 130
]

# Bounds for decision variables (based on data ranges)
bounds = [
    (df_smoothed['Melt_temp'].min(), df_smoothed['Melt_temp'].max()), 
    (df_smoothed['Mold_temp'].min(), df_smoothed['Mold_temp'].max()),  
    (df_smoothed['Fill_time'].min(), df_smoothed['Fill_time'].max()),  
    (df_smoothed['Plastic_time'].min(), df_smoothed['Plastic_time'].max()),  
    (df_smoothed['Closing_force'].min(), df_smoothed['Closing_force'].max()),  
    (df_smoothed['Inj_pressure_peak'].min(), df_smoothed['Inj_pressure_peak'].max())  
]
